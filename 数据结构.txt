登录信息
路由:http://127.0.0.1:8080/sign_up
请求方式:POST
前端发送信息:
SignUpInfo:{
    username:string
    password:string
}
后端返回值:
    成功:
    SignUpMessage:{
        message:"登录成功!"
        flag:true
    }
    失败:
    SignUpMessage:{
        message:"登录失败!"
        flag:false
    }

注册信息
路由:http://127.0.0.1:8080/sign_in
请求方式:POST
前端发送信息:
SignInInfo:{
    username:string
    password:string
    sex:string
    birthday:string
    country:string
    province:string
    city:string
}
后端返回值:
    成功:
    SignInInfoMessage:{
        message:"获取用户信息成功!"
        flag:true
    }
    失败:
    SignInInfoMessage:{
        message:"获取用户信息失败!"
        flag:false
    }

请求用户信息
路由:http://127.0.0.1:8080/api/get_user_info
请求方式:GET
前端发送信息:
    SearchUserInfo{
        username:string
    }
后端发送信息:
    成功找到用户信息:
    UserInfo:{
        username:string
        password:string
        sex:string
        birthday:string
        country:string
        province:string
        city:string
    }
    失败:
    SearchUserInfoMessage:{
        message:"用户不存在!"
        flag:false
    }
